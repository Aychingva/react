#include<stdlib.h>
#include <time.h>

int dive(){
    return rand()%2+1;
}


int goal(int kick_direction,int dive_direction){
    return kick_direction!=dive_direction;
}


void print_ball(int ball_direction){
    if(ball_direction==1){
        puts("-----------------");
        
        puts("|    O          |");
        puts("|               |");
        puts("|               |");
    
    }else if(ball_direction==2){
        puts("-----------------");
        puts("|          O    |");
        puts("|               |");
        puts("|               |");


    }

}

     if(goalkeeper==1){
        puts("-----------------");

    }

}
